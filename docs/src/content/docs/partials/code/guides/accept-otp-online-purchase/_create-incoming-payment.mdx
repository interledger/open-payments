let incoming_request = IncomingPaymentRequest {
wallet_address: retailer_wallet_address.id.clone(),
incoming_amount: Some(Amount { value: "140000".into(), asset_code: "MXN".into(), asset_scale: 2 }),

.incoming_payments()
.create(
