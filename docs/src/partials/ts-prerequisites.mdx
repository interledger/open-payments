import { LinkOut, Disclosure } from '@interledger/docs-design-system'

<Disclosure toggleText='Prerequisites' client:load>

- Node 18
- A package manager such as NPM or PNPM
- <LinkOut href='https://github.com/interledger/open-payments/tree/main/packages/open-payments'>
    Open Payments SDK
  </LinkOut>
- <LinkOut href='https://www.npmjs.com/package/tsx'>TSX</LinkOut>

**Additional configuration**

Add `"type": "module"` to `package.json`

Add the following to `tsconfig.json`

```json
{
  "compilerOptions": {
    "target": "ES2022",
    "module": "ES2022"
  }
}
```

</Disclosure>
